  `mid` VARCHAR(50),
  `password` VARCHAR(32) NOT NULL,
  `name` VARCHAR(30) NOT NULL,
  PRIMARY KEY (`mid`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;



DROP TABLE IF EXISTS `dept`;

	DEPTNO int(2),
	DNAME VARCHAR(14) ,
